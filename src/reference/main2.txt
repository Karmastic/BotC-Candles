#include <WiFi.h>
#include <WebSocketsClient.h> // Library for connecting via WebSocket.

const char* ssid = "";         // Replace with your network SSID
const char* password = ""; // Replace with your network password
const char* host = "echo.websocket.org"; // WebSocket server host
const uint16_t port = 80;               // Port for the server (use 443 for SSL)

WebSocketsClient webSocket;
unsigned long previousMillis = 0;
const long interval = 5000; // Send a message every 5 seconds

// WebSocket event handler function
void onWebSocketEvent(WStype_t type, uint8_t* payload, size_t length) {
    switch (type) {
        case WStype_DISCONNECTED:
            Serial.println("[WS] Disconnected");
            break;
        case WStype_CONNECTED:
            Serial.printf("[WS] Connected to url: %s\n", payload);
            // Send a message immediately upon connection
            webSocket.sendTXT("Hello World!");
            break;
        case WStype_TEXT:
            Serial.printf("[WS] Received message: %s\n", payload);
            break;
        case WStype_BIN:
            Serial.println("[WS] Received binary message");
            break;
        default:
            break;
    }
}

void setup() {
    Serial.begin(115200);
    delay(1000);
    Serial.println();
    Serial.print("Connecting to ");
    Serial.println(ssid);

    // Connect to Wi-Fi
    WiFi.begin(ssid, password);
    while (WiFi.status() != WL_CONNECTED) {
        delay(500);
        Serial.print(".");
    }
    Serial.println("\nWiFi connected");
    Serial.print("IP address: ");
    Serial.println(WiFi.localIP());

    // Connect to WebSocket server
    // Use webSocket.beginSSL() for wss:// connections and port 443
    webSocket.begin(host, port, "/"); 
    webSocket.onEvent(onWebSocketEvent);
    webSocket.setReconnectInterval(5000); // Attempt to reconnect every 5s if disconnected
}

void loop() {
    webSocket.loop(); // Required for the library to process events

    unsigned long currentMillis = millis();
    if (currentMillis - previousMillis >= interval) {
        previousMillis = currentMillis;
        // Send a test message periodically
        webSocket.sendTXT("Info to be echoed back");
    }
}

    case WStype_DISCONNECTED:                         // if a client is disconnected, then type == WStype_DISCONNECTED

      // try to decipher the JSON string received

        Serial.print(F("deserializeJson() failed: "));
